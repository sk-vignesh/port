# Plan 06-01: XIRR/CAGR Math Library + Enhanced Gains

## Objective
Build a pure `lib/xirr.ts` that implements the Newton-Raphson XIRR algorithm (the same one Excel uses). Enhance the `gains` Edge Function to compute and return CAGR and XIRR per holding alongside the existing P&L, using each holding's chronological buy/sell cashflow series.

## Context
The `gains` Edge Function (167 lines) already:
- Builds unrealised and realised P&L per security
- Fetches latest prices from `price_history`
- Returns `holdings[]` with `costBasis`, `currentValue`, `unrealizedGain`, `realizedGain`

Missing: no XIRR or CAGR. Both need the cashflow timeline (buy date → amount, sell date → amount).
The `allTxns` array is already available (ordered oldest first) — we just need to add the XIRR computation layer.

### XIRR Algorithm (Newton-Raphson)
```
XIRR({cashflows: [{date, amount}]}) → annualRate

NPV(r, cashflows) = Σ Ci / (1 + r)^(di/365)
  where di = days since first cashflow

Newton-Raphson:
  r_next = r - NPV(r) / NPV'(r)  (repeat until |NPV| < threshold)
```

CAGR can be approximated as:
```
CAGR = (currentValue / costBasis)^(1/years) - 1
  where years = daysBetween(earliest_buy_date, today) / 365
```

Create `lib/xirr.ts` with both algorithms:

```ts
/**
 * lib/xirr.ts — XIRR and CAGR calculation utilities
 *
 * XIRR: Extended Internal Rate of Return (Newton-Raphson, identical to Excel XIRR)
 * CAGR: Compound Annual Growth Rate (simple approximation)
 *
 * All dates as ISO YYYY-MM-DD strings.
 * All monetary values in any consistent unit (BIGINT×100 is fine).
 */

export interface Cashflow {
  date:   string   // ISO YYYY-MM-DD
  amount: number   // negative = outflow (buy), positive = inflow (sell / current value)
}

/** Days between two ISO date strings */
function days(from: string, to: string): number {
  return (new Date(to).getTime() - new Date(from).getTime()) / 86_400_000
}

/** NPV of cashflows at rate r, relative to first cashflow date */
function npv(rate: number, flows: Cashflow[]): number {
  const t0 = flows[0].date
  return flows.reduce((sum, f) => sum + f.amount / Math.pow(1 + rate, days(t0, f.date) / 365), 0)
}

/** Derivative of NPV w.r.t. rate */
function dnpv(rate: number, flows: Cashflow[]): number {
  const t0 = flows[0].date
  return flows.reduce((sum, f) => {
    const d = days(t0, f.date) / 365
    return sum - (d * f.amount) / Math.pow(1 + rate, d + 1)
  }, 0)
}

/**
 * Extended IRR (XIRR) via Newton-Raphson.
 *
 * @param flows  Array of cashflows — at least one negative (outflow) and one positive (inflow)
 * @param guess  Initial guess (default: 0.1 = 10%)
 * @returns      Annual rate as a decimal (0.15 = 15%), or null if no convergence
 */
export function xirr(flows: Cashflow[], guess = 0.1): number | null {
  if (flows.length < 2) return null
  let r = guess
  for (let i = 0; i < 100; i++) {
    const f  = npv(r, flows)
    const df = dnpv(r, flows)
    if (Math.abs(df) < 1e-12) return null
    const next = r - f / df
    if (Math.abs(next - r) < 1e-7) return next
    r = next
    // Clamp to avoid divergence
    if (r < -0.999) r = -0.999
    if (r > 100)    r = 100
  }
  return null   // Did not converge
}

/**
 * Simple CAGR: (endValue / startValue)^(1 / years) - 1
 *
 * @returns Annual rate as a decimal (0.15 = 15%), or null if insufficient data
 */
export function cagr(startValue: number, endValue: number, startDate: string, endDate: string): number | null {
  if (startValue <= 0 || endValue <= 0) return null
  const d = days(startDate, endDate) / 365
  if (d <= 0) return null
  return Math.pow(endValue / startValue, 1 / d) - 1
}
```
