```tsx
'use client'

/**
 * PerformerCards — Fetches from the gains Edge Function on mount
 * and displays Best Performer + Worst Performer callout cards.
 */
import { useState, useEffect } from 'react'
import { formatAmount } from '@/lib/format'

const SUPABASE_URL = process.env.NEXT_PUBLIC_SUPABASE_URL ?? ''
const ANON_KEY    = process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY ?? ''

interface GainsRow {
  name: string; ticker: string | null; currency: string
  currentValue: number | null; unrealizedGain: number | null
  unrealizedGainPct: number | null; xirr: number | null
}

export default function PerformerCards() {
  const [best,  setBest]  = useState<GainsRow | null>(null)
  const [worst, setWorst] = useState<GainsRow | null>(null)
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    async function fetchGains() {
      try {
        const { createClient } = await import('@/lib/supabase/client')
        const supabase = createClient()
        const { data: { session } } = await supabase.auth.getSession()
        if (!session) { setLoading(false); return }

        const res = await fetch(`${SUPABASE_URL}/functions/v1/gains`, {
          headers: { Authorization: `Bearer ${session.access_token}`, apikey: ANON_KEY },
        })
        const { holdings } = await res.json()
        if (!Array.isArray(holdings) || holdings.length === 0) { setLoading(false); return }

        // Only consider holdings with a current price
        const priced = holdings.filter((h: GainsRow) => h.currentValue != null && h.unrealizedGainPct != null)
        if (priced.length === 0) { setLoading(false); return }

        const sorted = [...priced].sort((a: GainsRow, b: GainsRow) =>
          (b.unrealizedGainPct ?? 0) - (a.unrealizedGainPct ?? 0)
        )
        setBest(sorted[0])
        setWorst(sorted[sorted.length - 1])
      } catch { /* silent */ }
      setLoading(false)
    }
    fetchGains()
  }, [])

  if (loading || (!best && !worst)) return null

  const Card = ({ row, label, isPositive }: { row: GainsRow; label: string; isPositive: boolean }) => (
    <div style={{
      padding: '16px 18px', borderRadius: 12,
      border: `1px solid ${isPositive ? 'rgba(34,197,94,0.25)' : 'rgba(239,68,68,0.25)'}`,
      background: isPositive ? 'rgba(34,197,94,0.05)' : 'rgba(239,68,68,0.05)',
      flex: 1, minWidth: 200,
    }}>
      <div style={{ fontSize: '0.6rem', fontWeight: 700, textTransform: 'uppercase', letterSpacing: '0.08em', color: isPositive ? '#22c55e' : '#ef4444', marginBottom: 8 }}>
        {label}
      </div>
      <div style={{ fontWeight: 700, fontSize: '0.92rem', marginBottom: 4 }}>{row.name}</div>
      {row.ticker && <div style={{ fontSize: '0.7rem', color: 'var(--color-text-muted)', marginBottom: 10 }}>{row.ticker}</div>}
      <div style={{ fontSize: '1.4rem', fontWeight: 800, color: isPositive ? '#22c55e' : '#ef4444' }}>
        {isPositive ? '+' : ''}{((row.unrealizedGainPct ?? 0) * 100).toFixed(1)}%
      </div>
      {row.unrealizedGain != null && (
        <div style={{ fontSize: '0.78rem', color: 'var(--color-text-muted)', marginTop: 2 }}>
          {isPositive ? '+' : ''}{formatAmount(row.unrealizedGain, row.currency)}
        </div>
      )}
      {row.xirr != null && (
        <div style={{ fontSize: '0.72rem', color: 'var(--color-text-muted)', marginTop: 6 }}>
          XIRR: <strong style={{ color: 'var(--color-text-primary)' }}>{(row.xirr * 100).toFixed(2)}%</strong>
        </div>
      )}
    </div>
  )

  return (
    <div style={{ display: 'flex', gap: 14, marginBottom: 20, flexWrap: 'wrap' }}>
      {best  && <Card row={best}  label="🏆 Best Performer"  isPositive />}
      {worst && <Card row={worst} label="⚠ Worst Performer" isPositive={false} />}
    </div>
  )
}
```

Then in `app/(app)/page.tsx`, import and render `PerformerCards` above the chart section.
</action>
