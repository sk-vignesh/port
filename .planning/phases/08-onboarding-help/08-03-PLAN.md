# Plan 08-03: In-App PageHelp System

## Objective
Add a `?` help button to every page header that opens a right-side drawer explaining what the page does, how to use it, and tips — so new users are never lost.

## Architecture

### Component: `components/PageHelp.tsx`
A self-contained client component that:
- Renders a `?` circular button (always visible in top-right of page header)
- On click, opens a slide-in drawer from the right (320px wide)
- Drawer has a close button, title, and scrollable content
- Drawer content is determined by `pageId` prop (looked up in `lib/pageHelp.ts`)
- Uses CSS `transform: translateX` for the slide animation (no dependencies)

```tsx
<PageHelp pageId="dashboard" />
// or
<PageHelp pageId="gains" />
```

### Drawer UI (no library dependencies)

```tsx
export default function PageHelp({ pageId }: { pageId: string }) {
  const [open, setOpen] = useState(false)
  const help = PAGE_HELP[pageId]
  if (!help) return null

  return (
    <>
      {/* ? Button */}
      <button onClick={() => setOpen(true)} title="Help for this page" style={{
        width: 28, height: 28, borderRadius: '50%', border: '1px solid var(--color-border)',
        background: 'var(--color-bg-elevated)', color: 'var(--color-text-muted)',
        fontSize: '0.78rem', fontWeight: 700, cursor: 'pointer',
        display: 'flex', alignItems: 'center', justifyContent: 'center',
        flexShrink: 0,
      }}>?</button>

      {/* Backdrop */}
      {open && (
        <div onClick={() => setOpen(false)} style={{
          position: 'fixed', inset: 0, background: 'rgba(0,0,0,0.3)', zIndex: 1000,
        }} />
      )}

      {/* Drawer */}
      <div style={{
        position: 'fixed', top: 0, right: 0, width: 320, height: '100vh',
        background: 'var(--color-bg-elevated)', borderLeft: '1px solid var(--color-border)',
        zIndex: 1001, transform: open ? 'translateX(0)' : 'translateX(100%)',
        transition: 'transform 0.25s cubic-bezier(.4,0,.2,1)',
        display: 'flex', flexDirection: 'column', overflowY: 'auto', padding: '24px 20px',
      }}>
        <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', marginBottom: 20 }}>
          <div style={{ fontWeight: 800, fontSize: '1rem' }}>{help.title}</div>
          <button onClick={() => setOpen(false)} style={{ background: 'none', border: 'none', cursor: 'pointer', color: 'var(--color-text-muted)', fontSize: '1.2rem' }}>✕</button>
        </div>
        <div style={{ display: 'flex', flexDirection: 'column', gap: 20 }}>
          {help.content.map((s, i) => (
            <div key={i}>
              <div style={{ fontWeight: 700, fontSize: '0.82rem', marginBottom: 6, color: 'var(--color-accent-light)' }}>{s.heading}</div>
              <div style={{ fontSize: '0.82rem', lineHeight: 1.7, color: 'var(--color-text-secondary)' }}>{s.text}</div>
            </div>
          ))}
        </div>
        <div style={{ marginTop: 'auto', paddingTop: 20, borderTop: '1px solid var(--color-border)', fontSize: '0.72rem', color: 'var(--color-text-muted)' }}>
          Need more detail? <a href="/help" style={{ color: 'var(--color-accent-light)' }}>Open the full Help Centre →</a>
        </div>
      </div>
    </>
  )
}
```
